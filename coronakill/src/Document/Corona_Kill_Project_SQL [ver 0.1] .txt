CREATE TABLE `member_info` (
  `member_id` varchar(50) NOT NULL,
  `member_password` varchar(50) NOT NULL,
  `member_name` varchar(50) NOT NULL,
  `member_adress` varchar(50) NOT NULL,
  `member_rank` varchar(50) NOT NULL,
  `member_number` varchar(50) NOT NULL,
  `member_result` varchar(50) DEFAULT NULL,
  `member_selfresult` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`member_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `hospital_info` (
  `hospital_id` varchar(50) NOT NULL,
  `hospital_adress` varchar(50) NOT NULL,
  `hospital_name` varchar(50) NOT NULL,
  `hospital_type` varchar(50) NOT NULL,
  `hospital_time` varchar(50) NOT NULL,
  `hospital_password` varchar(50) NOT NULL,
  `hospital_number` varchar(50) NOT NULL,
  PRIMARY KEY (`hospital_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `reservation` (
  `member_id` varchar(50) NOT NULL,
  `reservation_id` varchar(50) NOT NULL,
  `reservation_adr` varchar(50) NOT NULL,
  `reservation_hospital` varchar(50) NOT NULL,
  `reservation_date` varchar(50) NOT NULL,
  `reservation_type` varchar(50) NOT NULL,
  `hospital_id` varchar(50) NOT NULL,
  PRIMARY KEY (`reservation_id`),
  KEY `R_2` (`member_id`),
  KEY `R_3` (`hospital_id`),
  CONSTRAINT `reservation_ibfk_1` FOREIGN KEY (`member_id`) REFERENCES `member_info` (`member_id`),
  CONSTRAINT `reservation_ibfk_2` FOREIGN KEY (`hospital_id`) REFERENCES `hospital_info` (`hospital_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `board` (
  `board_id` varchar(50) NOT NULL,
  `board_title` varchar(50) NOT NULL,
  `board_content` varchar(50) NOT NULL,
  `board_date` varchar(50) NOT NULL,
  `member_id` varchar(50) NOT NULL,
  PRIMARY KEY (`board_id`),
  KEY `R_1` (`member_id`),
  CONSTRAINT `board_ibfk_1` FOREIGN KEY (`member_id`) REFERENCES `member_info` (`member_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
